	\def\pin{#2} % coordinate for centre of roller
	\def\ffill{#3} % fill color

    \def\width{#5} % stroke width in pt

        \filldraw[rounded corners=0.5mm] ($(\pin) + (-0.7,-1)$) arc(180:90:.2) -- ++(0.05,0)arc(-90:0:0.2) -- ++(0.05,0.4)arc(225:-45:0.28284)-- ++(0.05,-.4)arc(180:270:.2)-- ++(0.05,0)arc(90:0:0.2) -- cycle;

        \fill[outer color=\ffill, inner color=black] (\pin) circle (2.5mm);
		\filldraw[fill=white] (\pin) circle (1.25mm);
